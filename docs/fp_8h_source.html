202struct __attribute__((packed)) FujiBinaryProfile {
203 // So far this has been observed as 0x17 or 0x1d and seems to represent the number of properties following iop_codes.
204 // The problem is, the camera appears to always send this structure as exactly 601 bytes regardless of n_props.
205 // In that case, one would expect n_props to be 0x16, but it puts 0x17 instead.
206 // X Raw Studio seems to always write this correctly as 0x1d and sends 629 bytes.
207 // When this structure is read back from the camera, it again sends it back as 601 bytes with n_props being 0x1d.
208 uint16_t n_props;
209 // This seems to hold a series of (or just one) standard MTP strings.
210 // The camera also seems to incorrectly format this property, writing the string as length 8 rather than 9.
211 // (See MTP spec page 21 - length of a string includes the null terminator)
212 char iop_codes[0x1ff];
213 // (offset 0x201)
214 // These appear to be in the same order as the props in the XML files
215 union Props {
216 uint32_t values[0x1d];
